//flatten一个数组并且去重

function flatten(arr,outer){
	var len = arr.length;
	var ret = !outer? []:outer;

	for(var i = 0; i < len;i++){
		var temp = arr[i];
		if(typeof arr[i] == 'number'){
			ret.push(arr[i]);
		}else{
			flatten(temp,ret);
		}
	}

	return ret;
}

function unique(arr){
	var hash = {};
	var ret = [];

	for(var i = 0; i < arr.length; i++){
		var k = arr[i];
		if(!hash[k]){
			hash[k] = 1;
			ret.push(k);
		}
	}
	return ret;
}
